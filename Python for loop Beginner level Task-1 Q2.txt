total_jumpings = 100
size= 10
completed_jumpings= 0
remaining_jacks=0
while completed_jumpings < total_jumpings:
    completed_jumpings +=size
    remaining_jacks = total_jumpings- completed_jumpings 
    value = input(f"Are you tired after the  completing {completed_jumpings} (yes/y or no/n): ").lower()
    if value in ['yes', 'y']:
         skip = input("Do you want to skip the remaining sets? (yes/y or no/n):").lower()
         if skip=="yes":
             print(f"You completed a total of {completed_jumpings} jumping jacks.")
             break
    if completed_jumpings>= total_jumpings:
            print(" You Completed the workout.")
    else:
            print(f"{remaining_jacks} jumping jacks remaining.")

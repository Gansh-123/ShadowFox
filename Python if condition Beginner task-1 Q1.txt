# 4. if condition

height=float(input('Enter Your Height in meter:'))
weight=float(input('Enter Your weight in kilograms: '))
bmi=weight/(height**2)
print(bmi)
if bmi>=30:
    print('Obesity')
elif 25<bmi<29:
    print('Overweight')  
elif 18.5<bmi<25:
    print('Normal') 
elif bmi<18.5:
    print('Underweight')
